import java.util.Scanner; 
public class Main { 
    public static void main(String[] args) {
        
        Scanner in = new Scanner(System.in);
		
		int test = in.nextInt();
		
		for(int i=1; i<=test; i++) {
			int a = in.nextInt();
			int b = in.nextInt();
			System.out.printf("Case #"+i+": "+(a+b)+"\n");
		}
		
		in.close();
    }
}
